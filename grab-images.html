FF Image Downloader

For each listing below, open it in a new tab, paste the script in the browser console (Cmd+Option+J), and it will auto-download 5 images with the correct filenames.

Important: After downloading, move the images from your Downloads folder to la-rental-dashboard/images/

Step 2: Paste this in the browser console on EACH listing page

This finds the property images and triggers browser downloads with correct filenames.

(() => {
  const MAP = {
    '481770': 15, '941378': 16, '894445': 17, '850346': 18,
    '621424': 19, '914638': 20, '355699': 21, '752691': 22,
    '731030': 23, '815538': 24, '913477': 25
  };
  const propId = location.pathname.match(/\/property\/(\d+)/)?.[1] || '';
  const listingId = MAP[propId];
  if (!listingId) { alert('Unknown property ID: ' + propId); return; }

  // Find all property images (not avatars, logos, tracking pixels)
  const imgs = [...document.querySelectorAll('img')]
    .map(i => i.src || i.dataset?.src || '')
    .filter(s => s && s.includes('_pdp_/' + propId))
    .filter((v, i, a) => a.indexOf(v) === i)  // dedupe
    .slice(0, 5);

  if (!imgs.length) {
    alert('No property images found! Try scrolling through all photos first, then run again.');
    return;
  }

  alert('Found ' + imgs.length + ' images for Listing ' + listingId + '. Downloading now...');

  imgs.forEach((url, i) => {
    const ext = url.match(/\.(jpe?g|png|webp)/i)?.[0]?.toLowerCase() || '.jpg';
    const filename = 'listing_' + listingId + '_img_' + (i + 1) + ext;
    fetch(url)
      .then(r => r.blob())
      .then(blob => {
        const a = document.createElement('a');
        a.href = URL.createObjectURL(blob);
        a.download = filename;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(a.href);
        console.log('Downloaded: ' + filename);
      })
      .catch(e => console.error('Failed: ' + filename, e));
  });
})();
